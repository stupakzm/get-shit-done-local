```bash
# Before any Task() that uses a model variable from init:
if [[ "{some_model}" == ollama:* ]]; then
  OLLAMA_MODEL_NAME="${some_model#ollama:}"
  # Pipe the prompt content to gsd-tools ollama run
  # The prompt must be captured as a variable before this guard
  echo "$PROMPT_CONTENT" | node "$HOME/.claude/get-shit-done/bin/gsd-tools.cjs" ollama run "$OLLAMA_MODEL_NAME"
else
  Task(
    subagent_type="...",
    model="{some_model}",
    prompt="..."
  )
fi
```
